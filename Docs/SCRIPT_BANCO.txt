--CLIENTE--------------------------------------------------
CREATE TABLE CLIENTE (
ID_CLIENTE BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
-- >>> CAMPOS DE NEGÓCIO (EXEMPLOS)
NOME VARCHAR(120) NOT NULL,
RAZAO_SOCIAL VARCHAR(120),
CPF_CNPJ VARCHAR(14) NOT NULL,
TELEFONE VARCHAR(12),
TIPO_PESSOA CHAR(1) NOT NULL,
-- >>> CONTROLE
ATIVO SMALLINT DEFAULT 1 NOT NULL,
-- >>> AUDITORIA
DT_CADASTRO TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL
);

-- CHECKS
ALTER TABLE CLIENTE ADD CONSTRAINT CK_CLIENTE__ATIVO CHECK (ATIVO IN (0,1));
ALTER TABLE CLIENTE ADD CONSTRAINT CK_CLIENTE__TIPO_PESSOA CHECK (TIPO_PESSOA IN ('F','J'));
-- UNIQUE
ALTER TABLE CLIENTE ADD CONSTRAINT UK_CLIENTE__CPF_CNPJ UNIQUE (CPF_CNPJ);
-- ÍNDICES
CREATE INDEX IX_CLIENTE_NOME ON CLIENTE (NOME);
CREATE INDEX IX_CLIENTE_CPF_CNPJ ON CLIENTE (CPF_CNPJ);

--USUARIO--------------------------------------------------
CREATE TABLE USUARIO (
ID_USUARIO BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,

NOME VARCHAR(100) NOT NULL,
LOGIN VARCHAR(50) NOT NULL,
SENHA VARCHAR(100) NOT NULL,

ATIVO SMALLINT DEFAULT 1 NOT NULL,
DT_CADASTRO TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL
);

-- CHECKS
ALTER TABLE USUARIO ADD CONSTRAINT CK_USUARIO__ATIVO CHECK (ATIVO IN (0,1));
-- UNIQUE
ALTER TABLE USUARIO ADD CONSTRAINT UK_USUARIO__CPF_CNPJ UNIQUE (LOGIN);

--PERMISSAO USUARIO--------------------------------------------------
CREATE TABLE PERMISSAO (
ID_PERMISSAO BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,

NOME VARCHAR(50) NOT NULL,
DESCRICAO VARCHAR(200)
);

CREATE TABLE USUARIO_PERMISSAO (
ID_USUARIO BIGINT NOT NULL,
ID_PERMISSAO BIGINT NOT NULL,

PRIMARY KEY (ID_USUARIO, ID_PERMISSAO),

FOREIGN KEY (ID_USUARIO) REFERENCES USUARIO(ID_USUARIO),
FOREIGN KEY (ID_PERMISSAO) REFERENCES PERMISSAO(ID_PERMISSAO)
);
